--SE CREAN AMBOS CAMPOS EN LAS DOS TABLAS DE LA RELACION CON EL MISMO NOMBRE.


 --BORRADO DE TABLAS

DROP TABLE ROL_EMPLEADO;
DROP TABLE TIPO_CONTRATA;
DROP TABLE PAISES;
DROP TABLE DEPARTAMENTOS;
DROP TABLE CIUDADES;
DROP TABLE SEDES;
DROP TABLE FACULTADES;
DROP TABLE EMPLEADOS;
DROP TABLE AREAS;
DROP TABLE PROGRAMAS;

--CREACION DE TABLAS

CREATE TABLE PAISES (
CODIGO_PAIS INT, 
NOMBRE_PAIS VARCHAR(20) NOT NULL,
CONSTRAINT PK_PAISES PRIMARY KEY (CODIGO_PAIS)
);

CREATE TABLE DEPARTAMENTOS (
CODIGO_DEPTO INT, 
NOMBRE_DEPTO VARCHAR(20) NOT NULL,
CODIGO_PAIS INT,
CONSTRAINT PK_DEPTOS PRIMARY KEY (CODIGO_DEPTO),
CONSTRAINT FK_DEPTOSPAISES FOREIGN KEY (CODIGO_PAIS) REFERENCES PAISES(CODIGO_PAIS)
);

CREATE TABLE CIUDADES (
CODIGO_CIUDAD INT, 
NOMBRE_CIUDAD VARCHAR(20) NOT NULL,
CODIGO_DEPTO INT,
CONSTRAINT PK_CIUDADES PRIMARY KEY (CODIGO_CIUDAD),
CONSTRAINT FK_CIUDADESDEPTOS FOREIGN KEY (CODIGO_DEPTO) REFERENCES DEPARTAMENTOS(CODIGO_DEPTO)
);

CREATE TABLE SEDES (
CODIGO_SEDE INT, 
NOMBRE_SEDE VARCHAR(20) NOT NULL,
CODIGO_CIUDAD INT,
CONSTRAINT PK_SEDES PRIMARY KEY (CODIGO_SEDE),
CONSTRAINT FK_SEDESCIUDADES FOREIGN KEY (CODIGO_CIUDAD) REFERENCES CIUDADES(CODIGO_CIUDAD)
);

CREATE TABLE TIPO_CONTRATA (
NOMBRE_CONTRATACION VARCHAR(20) NOT NULL, 
CONSTRAINT PK_NOMBRECONTRATACION PRIMARY KEY (NOMBRE_CONTRATACION)
);

CREATE TABLE ROL_EMPLEADO (
NOMBRE_ROL VARCHAR(20) NOT NULL, 
CONSTRAINT PK_NOMBREROL PRIMARY KEY (NOMBRE_ROL)
);

CREATE TABLE FACULTADES (
CODIGO_FACULTAD INT, 
NOMBRE_FACULTAD VARCHAR(20) NOT NULL,
UBICA_FACULTAD	VARCHAR(15) NOT NULL,
NRO_TELEFONO	VARCHAR(15) NOT NULL,
ID_DECANO	VARCHAR(15) UNIQUE,
CONSTRAINT PK_FACULTADES PRIMARY KEY (CODIGO_FACULTAD)
);

CREATE TABLE EMPLEADOS (
IDENTIFICACION VARCHAR(15), 
NOMBRES VARCHAR(30) NOT NULL,
APELLIDOS VARCHAR(30) NOT NULL,
EMAIL VARCHAR(30) NOT NULL,
NOMBRE_CONTRATACION VARCHAR(20) NOT NULL,
NOMBRE_ROL VARCHAR(20) NOT NULL,
CODIGO_FACULTAD INT,
CODIGO_SEDE INT,
LUGAR_NACIMIENTO INT,
CONSTRAINT PK_EMPLEADOS PRIMARY KEY (IDENTIFICACION),
CONSTRAINT FK_EMPLEADOSCONTRATACION FOREIGN KEY (NOMBRE_CONTRATACION) REFERENCES TIPO_CONTRATA (NOMBRE_CONTRATACION),
CONSTRAINT FK_EMPLEADOSTIPOEM FOREIGN KEY (NOMBRE_ROL) REFERENCES ROL_EMPLEADO (NOMBRE_ROL),
CONSTRAINT FK_EMPLEADOSFACULTADES FOREIGN KEY (CODIGO_FACULTAD) REFERENCES FACULTADES(CODIGO_FACULTAD),
CONSTRAINT FK_EMPLEADOSSEDES FOREIGN KEY (CODIGO_SEDE) REFERENCES SEDES (CODIGO_SEDE),
CONSTRAINT FK_EMPLEADOSNACIMIENTO FOREIGN KEY (LUGAR_NACIMIENTO) REFERENCES CIUDADES (CODIGO_CIUDAD)
);

CREATE TABLE AREAS (
CODIGO_AREA INT, 
NOMBRE_AREA VARCHAR(20) NOT NULL,
CODIGO_FACULTAD INT,
ID_COORDINADOR VARCHAR(15) UNIQUE,
CONSTRAINT PK_AREAS PRIMARY KEY (CODIGO_AREA),
CONSTRAINT FK_AREASFACULTADES FOREIGN KEY (CODIGO_FACULTAD) REFERENCES FACULTADES(CODIGO_FACULTAD),
CONSTRAINT FK_AREASEMPLEADOS FOREIGN KEY (ID_COORDINADOR) REFERENCES EMPLEADOS(IDENTIFICACION)
);

CREATE TABLE PROGRAMAS (
CODIGO_PROGRAMA INT, 
NOMBRE_PROGRAMA VARCHAR(20) NOT NULL,
CODIGO_AREA INT,
CONSTRAINT PK_PROGRAMAS PRIMARY KEY (CODIGO_PROGRAMA),
CONSTRAINT FK_PROGRAMASAREAS FOREIGN KEY (CODIGO_AREA) REFERENCES AREAS(CODIGO_AREA)
);
